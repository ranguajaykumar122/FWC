\begin{enumerate}
	\item If \text{$\vec{a}, \vec{b} and \vec{c}$} are the position vectors of the points A$$(2,3,-4), B(3,-4,5) and C(3,2,-3)$$ respectively, then \text{$|\vec{a} + \vec{b} + \vec{c}|$} is equal to
\begin{enumerate}[label=(\Alph*)]
\item $\sqrt{113}$
\item $\sqrt{185}$
\item $\sqrt{203}$
\item $\sqrt{209}$
\end{enumerate}
\item Find the distance of the point $(a, b, c)$ from the x-axis.
\item \begin{enumerate} %[label=(\alph*)] 
		\item If \text{$\vec{a}= 2\hat{i} - \hat{j} + 2\hat{k}$} and \text{$\vec{b} = 5\hat{i} - 3\hat{j} - 4\hat{k}$}, then find the ratio
	\\ $\frac{projection of vector \vec{a} on vector \vec{b}}{projection of vector \vec{b} on vector \vec{a}}$
\item Let \text{$\hat{a}$ and $\hat{b}$} be two unit vectors. If the vectors 
	\begin{align}
		\text{$\vec{c} = \hat{a} + 2\hat{b}$}
	\end{align}
		and 
		\begin{align}
			\text{$\vec{d} = 5\hat{a} - 4\hat{b}$} 
		\end{align}
		are perpendicular to each other, then find the angle between the vectors \text{$\vec{a}$ and $\vec{b}.$} 
\end{enumerate}
\item Show that \text{$|\vec{a}|\vec{b} +|\vec{b}|\vec{a}$} is perpendicular to \text{$|\vec{a} |\vec{b} - |\vec{b}|\vec{a}$}, for any two non-zero vectors \text{$\vec{a}$ and $\vec{b}.$}
\item Prove that three points A, B and C with position vectors $\vec{a}$, $\vec{b}$ and $\vec{c}$ respectively are collinear if and only if 
	\begin{align}
		(\vec{b} \times \vec{c}) + (\vec{c} \times \vec{a}) + (\vec{a} \times \vec{b}) = 0
	\end{align}
\end{enumerate}
